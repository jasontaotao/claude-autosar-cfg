### 2.2 New flow (parity with CLI)

```ts
applyMutation: async (): Promise<void> => {
  const result = get().runResult;
  if (result === null) return;

  const { useArxmlStore } = await import('./useArxmlStore.js');
  const { applyPatchSteps } = await import('../../core/mutation/applyPatchSteps.js');
  const { serializeArxml } = await import('../../core/arxml/serializer.js');
  const { scriptMutationToPatchStep } = await import('./helpers/scriptMutationToPatchStep.js');

  const arxmlState = useArxmlStore.getState();
  const filePath = arxmlState.activeDocumentPath;

  // No active doc ŌåÆ nothing to apply. Preserve stub contract.
  if (arxmlState.doc === null || filePath === null) {
    set({ runResult: { ...result, mutations: [] }, dirty: false });
    return;
  }

  // Build PatchStep[] from ScriptMutation[]. Mapper is a pure function.
  const steps = result.mutations.map(scriptMutationToPatchStep);

  // Resolve moduleDef from bswmdSchemas for BSWMD-aware add-child.
  const moduleDef = resolveModuleDefForActiveDoc(arxmlState);

  // Apply via the same engine the CLI uses.
  const applyResult = applyPatchSteps(arxmlState.doc, steps, { moduleDef });

  // Thread warnings back into runResult.warnings.
  const warnings: ScriptStepWarning[] = applyResult.warnings.map((w) => ({
    stepIndex: w.stepIndex,
    kind: w.kind,
    message: w.message,
  }));

  // Aggregate errors.
  const errorMessage =
    applyResult.errors.length > 0
      ? `${applyResult.applied}/${result.mutations.length} mutations applied; ${applyResult.errors.map((e) => `${e.kind}: ${e.message}`).join('; ')}`
      : undefined;

  if (applyResult.errors.length > 0) {
    set({
      runResult: {
        ...result,
        mutations: [],
        warnings,
        ...(errorMessage !== undefined ? { errorMessage } : {}),
      },
      dirty: false,
    });
    return;
  }

  // Update store doc via the canonical setDoc path (triggers displayDoc
  // recompute + validation refresh + dirty flag).
  useArxmlStore.setDoc(applyResult.doc, filePath);

  // Persist via project:save IPC.
  const serialized = serializeArxml(applyResult.doc);
  if (!serialized.ok) {
    set({
      runResult: {
        ...result,
        mutations: [],
        warnings,
        ...(errorMessage !== undefined
          ? { errorMessage: `serialize: ${serialized.error.message}` }
          : {}),
      },
      dirty: false,
    });
    return;
  }

  if (arxmlState.project === null || arxmlState.projectPath === null) {
    set({
      runResult: {
        ...result,
        mutations: [],
        warnings,
        ...(errorMessage !== undefined ? { errorMessage } : {}),
        errorMessage:
          errorMessage ?? 'project save skipped: script commit requires a loaded project',
      },
      dirty: false,
    });
    return;
  }

  const saveResult = await window.autosarApi.projectSave({
    manifestPath: arxmlState.projectPath,
    manifest: arxmlState.project,
    files: [{ path: filePath, content: serialized.value }],
  });
  if (saveResult.kind === 'write-failed') {
    set({
      runResult: {
        ...result,
        mutations: [],
        warnings,
        errorMessage: `projectSave: ${saveResult.message}`,
      },
      dirty: true,
    });
    return;
  }

  set({
    runResult: {
      ...result,
      mutations: [],
      warnings,
    },
    dirty: false,
  });
};
```

### 2.3 Mapping function: `scriptMutationToPatchStep`

NEW file `src/renderer/store/helpers/scriptMutationToPatchStep.ts`:

```ts
import type { PatchStep } from '../../core/mutation/applyPatchSteps.js';
import type { ScriptMutation } from '@shared/script/types.js';

/**
 * v1.20.0 T1 C2.4 ŌĆö map a script-shaped ScriptMutation to the
 * wire-shape PatchStep. The script engine uses 3 kinds (set-param /
 * add-child / remove-child) and looser type names; the CLI patch
 * engine uses 7 kinds (add/remove/replace + 3 AUTOSAR extensions)
 * with stricter types. The mapper is a thin field-rename + a
 * cascade-default flip.
 *
 *  - `set-param` ŌåÆ `set-param` (1:1, field rename: newValue ŌåÆ value)
 *  - `add-child` ŌåÆ `add-child` (1:1, field rename: containerPath/newShortName ŌåÆ parentPath/shortName)
 *  - `remove-child` ŌåÆ `remove-with-cascade { cascade: true }` (script commits cannot present the cascade dialog mid-script, so cascade MUST be true)
 */
export function scriptMutationToPatchStep(m: ScriptMutation): PatchStep {
  switch (m.kind) {
    case 'set-param':
      return {
        op: 'set-param',
        containerPath: m.containerPath,
        paramName: m.paramName,
        value: m.newValue as string | number | boolean | null,
      };
    case 'add-child':
      return {
        op: 'add-child',
        parentPath: m.containerPath,
        shortName: m.newShortName,
      };
    case 'remove-child':
      return {
        op: 'remove-with-cascade',
        containerPath: m.containerPath,
        cascade: true,
      };
  }
}
```

### 2.5 `resolveModuleDefForActiveDoc` helper

NEW file `src/renderer/store/helpers/resolveModuleDefForActiveDoc.ts`:

```ts
import type { ArxmlState } from '../useArxmlStore.js';
import type { BswModuleDef } from '../../core/project/bswmd.js';

/**
 * v1.20.0 T1 C2.4 ŌĆö resolve the BSWMD for the active doc's primary module.
 * Used by `applyMutation` to thread `moduleDef` to `applyPatchSteps` so
 * `add-child` ops get BSWMD-aware multiplicity + child-def lookup.
 *
 * Returns `undefined` when the doc has no module element, when the
 * BSWMD is not loaded, or when the module's shortName doesn't match
 * any loaded BSWMD. Callers pass through to `applyPatchSteps` which
 * emits `no-bswmd-for-module` for `add-child` when context is absent.
 */
export function resolveModuleDefForActiveDoc(state: ArxmlState): BswModuleDef | undefined {
  if (state.doc === null) return undefined;
  const moduleEl = state.doc.packages.flatMap((p) => p.elements).find((e) => e.kind === 'module');
  if (moduleEl === undefined) return undefined;
  return state.bswmdSchemas.find((s) => s.shortName === moduleEl.shortName);
}
```
