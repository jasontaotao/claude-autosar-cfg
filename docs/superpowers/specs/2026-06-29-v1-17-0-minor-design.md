```ts
// ACTUAL interface is at src/core/project/bswmd.ts:168-174 — the per-container
// schema-level reference. The existing `destKind: string` already captures the
// schema-level destination (e.g. `ECUC-REFERENCE-DEF`, `ECUC-FOREIGN-REFERENCE-DEF`),

// the dest belongs to so renderer + generator can route render/emit logic without
// parsing dest shape. Note: `ReferenceEdge` (src/core/generator/emit/reference.ts)
// is a different type — generator-pipeline cross-module ref post-validation.
// C10 attaches to the schema-level ReferenceDef, NOT to ReferenceEdge.
export interface ReferenceDef {

  readonly shortName: string;
  readonly path: string;
  readonly destKind: string;
  readonly lowerMultiplicity: number;
  readonly upperMultiplicity: number | 'infinite';

  // NEW (C10): cross-dialect discriminator — distinguishes which
  // dialect the dest comes from so the renderer + generator can
  // route render / emit logic correctly without parsing dest shape.
  readonly destDialect?: 'P-PORT' | 'R-PORT' | 'SW-C' | 'ECUC-MODULE-DEF';

// src/core/project/bswmd.ts — extend BswmdDocument with moduleRefs? (C11, ACTUAL TYPE not EcucDefinitionCollection)
// ACTUAL interface is at src/core/project/bswmd.ts:32-43. Top-level parsed document
// — has modules[], warnings[], version, disabledModules?. NO top-level references[].
// Currently <MODULE-REF> elements are not parsed at all by the existing parser
// (the parser only walks AR-PACKAGES/ELEMENTS for module defs). C11 ADDS
// moduleRefs collection by extending the parser to look at <MODULE-REF>
// children inside the same AR-PACKAGES tree and surfacing them at the document
// level. No parallel `EcucDefinitionCollection` type is introduced — the natural
// surface is the existing BswmdDocument.
export interface BswmdDocument {
  // ... existing fields preserved unchanged ...
  readonly version: string;
  readonly modules: readonly BswModuleDef[];
  readonly warnings: readonly string[];
  readonly disabledModules?: ReadonlySet<string>;

  // NEW (C11): explicit MODULE-REF attachment — these were silently
  // dropped by the parser pre-v1.17.0. Now they survive parsing and
  // are available to renderer tree + future generator work.
  readonly moduleRefs?: readonly ModuleRefEntry[];
}

// NEW (C11): new type co-located with BswmdDocument for parser clarity.
export interface ModuleRefEntry {
  readonly target: string; // path to module def in the BSWMD
  readonly source: string; // path of the referencing package (debug)
```

```tsx
// src/renderer/components/ErrorBoundary.tsx (NEW, PB-4)
interface ErrorBoundaryProps {
  readonly fallback?: (error: Error, reset: () => void) => JSX.Element;
  readonly children: React.ReactNode;
}

interface ErrorBoundaryState {
  readonly error: Error | null;
}

export class ErrorBoundary extends React.Component<ErrorBoundaryProps, ErrorBoundaryState> {
  state: ErrorBoundaryState = { error: null };

  static getDerivedStateFromError(error: Error): ErrorBoundaryState {
    return { error };
  }

  componentDidCatch(error: Error, info: React.ErrorInfo): void {
    // Telemetry hook for crash reporting
    console.error('[renderer ErrorBoundary]', error, info);
  }

  reset = (): void => this.setState({ error: null });

  render(): React.ReactNode {
    if (this.state.error !== null) {
      const { fallback } = this.props;
      return fallback ? (
        fallback(this.state.error, this.reset)
      ) : (
        <DefaultErrorFallback error={this.state.error} onReload={this.reset} />
      );
    }
    return this.props.children;
  }
}

function DefaultErrorFallback({
  error,
  onReload,
}: {
  error: Error;
  onReload: () => void;
}): JSX.Element {
  return (
    <div role="alert" className="error-boundary">
      <h1>Something went wrong</h1>
      <p>{error.message}</p>
      <button type="button" onClick={onReload}>
        Reload Renderer
      </button>
    </div>
  );
}
```

```ts
export function trackWrite(): () => void {
  inflightWrites += 1;
  return () => { inflightWrites -= 1; };

        app.exit(0);
```
