```ts
import { describe, it, expect } from 'vitest';
import { EcuCGenerator } from '../modules/ecuc.js';
import { McuGenerator } from '../modules/mcu.js';
import type { BswmdModuleDefLike } from '../../project/bswmd.js';
import type { EcucValuesContainerLike } from '../modules/ecuc.js';
import type { McuValuesContainerLike } from '../modules/mcu.js';
import type { GenerationContext } from '../registry.js';

function makeBswmdIndex(shortName: string, moduleHeader: string | undefined): ReadonlyMap<string, BswmdModuleDefLike> {
  return new Map([
    [shortName, { shortName, ...(moduleHeader !== undefined ? { moduleHeader } : {}) } as BswmdModuleDefLike],
  ]);
}

function makeEcuCDef(shortName: string): BswmdModuleDefLike {
  return {
    shortName,
    containers: [
      {
        shortName: 'PartitionConfig',
        parameters: [{ shortName: 'PartitionId', kind: 'integer' }],
      },
    ],
  } as unknown as BswmdModuleDefLike;
}

function makeMcuDef(shortName: string): BswmdModuleDefLike {
  return {
    shortName,
    containers: [
      {
        shortName: 'McuModuleConfiguration',
        parameters: [{ shortName: 'McuClockReferencePointFrequency', kind: 'integer' }],
      },
    ],
  } as unknown as BswmdModuleDefLike;
}

function makeCtx(bswmdIndex: ReadonlyMap<string, BswmdModuleDefLike>): GenerationContext {
  return {
    bswmdIndex: bswmdIndex as unknown as ReadonlyMap<string, unknown>,
    bswmdParamIndex: new Map(),
    outDir: '/tmp/out',
    diagnostics: [],
  };
}

describe('moduleHeader thread — BSWMD <HEADER> flows into template context', () => {
  it('Mcu: BSWMD <HEADER> overrides convention', () => {
    const def = makeMcuDef('Mcu');
    const values: McuValuesContainerLike = {
      references: [],
      containers: [
        { shortName: 'McuModuleConfiguration', parameters: [] },
      ],
    };
    const ctx = makeCtx(makeBswmdIndex('Mcu', 'Mcu/McuAlt_Cfg.h'));

    const result = McuGenerator.emit(def, values, ctx);

    // The .c source artifact should reference McuAlt_Cfg.h, not the
    // conventional Mcu/Mcu_Cfg.h. We assert on the source content
    // (post-template render).
    const cArtifact = result.artifacts.find((a) => a.path.endsWith('.c'));
    expect(cArtifact?.content).toContain('McuAlt_Cfg.h');
    expect(cArtifact?.content).not.toContain('"Mcu/Mcu_Cfg.h"');
  });

  it('EcuC: BSWMD <HEADER> overrides convention', () => {
    const def = makeEcuCDef('EcuC');
    const values: EcucValuesContainerLike = {
      references: [],
      containers: [
        { shortName: 'PartitionConfig', parameters: [{ shortName: 'PartitionId', value: 0 }] },
      ],
    };
    const ctx = makeCtx(makeBswmdIndex('EcuC', 'EcuC/EcuCAlt_Cfg.h'));

    const result = EcuCGenerator.emit(def, values, ctx);

    const hArtifact = result.artifacts.find((a) => a.path.endsWith('.h'));
    expect(hArtifact?.content).toContain('EcuCAlt_Cfg.h');
    expect(hArtifact?.content).not.toContain('"EcuC/EcuC_Cfg.h"');
  });
});
```
