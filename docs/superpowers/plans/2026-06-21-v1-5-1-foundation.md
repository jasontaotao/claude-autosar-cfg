```ts
// shared/paths/isPathInside.ts
// Pure path-containment check. Returns true iff `child` is strictly
// inside `parent` (i.e. `child !== parent`). Hardened against:
//   - path traversal via `..` segments
//   - trailing slashes
//   - Windows case-insensitivity
//   - UNC paths (\\server\share\...)
//   - current-directory marker (.)
//   - double-slash normalization
//
// Used by `src/main/ipc/register.ts` (project-save handler) and
// available to core/ for future use (e.g. resource scoping).

import { isAbsolute, relative } from 'node:path';

/**
 * Returns true iff `child` is strictly inside `parent`.
 * Both paths are normalized before comparison.
 */
export function isPathInside(child: string, parent: string): boolean {
  // Normalize trailing slashes
  const normChild = stripTrailingSep(child);
  const normParent = stripTrailingSep(parent);

  // Different drive on Windows → path.relative returns absolute path
  const rel = relative(normParent, normChild);

  // Empty rel means child === parent
  if (rel === '') return false;

  // Absolute rel means different drive (Windows) → not inside
  if (isAbsolute(rel)) return false;

  // Path traversal: starts with .. → not inside
  if (rel === '..' || rel.startsWith(`..${sep()}`)) return false;

  // Case-insensitive on Windows
  if (process.platform === 'win32') {
    return !rel.toLowerCase().startsWith(`..${sep()}`.toLowerCase()) && rel.toLowerCase() !== '..';
  }

  return true;
}

function stripTrailingSep(p: string): string {
  if (p.length === 0) return p;
  const last = p.charAt(p.length - 1);
  // Both POSIX (/) and Windows (\) separators
  if (last === '/' || last === '\\') {
    return p.slice(0, -1);
  }
  return p;
}

function sep(): string {
  // Use forward slash for our internal checks; path.relative handles
  // platform-specific separators, but our `..` check is string-based
  return '/';
}
```

```ts
/**
 * Serialize the document in the same order as the source XML. Used to
 * preserve user-edited ordering across save/reload. Tolerance rules
 * (Q5 B): namespace prefix order, whitespace, comments, attribute order
 * are not preserved — only the order of <AR-PACKAGE> / <ELEMENT> / <MODULE>
 * containers.
 */
function serializeInSourceOrder(
  doc: ArxmlDocument,
  sourceArxml: string,
): Result<string, SerializeError> {
  // Extract order from source: list of all AR-PACKAGE / ELEMENT / MODULE
  // short-name paths in document order.
  const sourceOrder = extractContainerOrder(sourceArxml);

  // Walk doc.packages in source order; for each, recursively emit
  // children in their source order. Containers not in source (newly
  // added) go to the end in their internal order.
  const ordered = reorderBySource(doc, sourceOrder);

  // Now serialize using the same fast-xml-builder path, but with the
  // reordered document.
  return serializeViaBuilder(ordered);
}

function extractContainerOrder(xml: string): readonly string[] {
  const paths: string[] = [];
  // Match the SHORT-NAME within <AR-PACKAGE>, <ELEMENT>, <ECUC-MODULE-DEF-VALUES>
  // Pattern is intentionally simple; relies on well-formed source XML.
  const re =
    /<(?:AR-PACKAGE|ELEMENT|ECUC-MODULE-DEF-VALUES|ECUC-MODULE-CONFIGURATION-VALUES)>\s*<SHORT-NAME>([^<]+)<\/SHORT-NAME>/g;
  let m: RegExpExecArray | null;
  while ((m = re.exec(xml)) !== null) {
    paths.push(m[1]!);
  }
  return paths;
}

function reorderBySource(doc: ArxmlDocument, sourceOrder: readonly string[]): ArxmlDocument {
  // Stable sort: containers matching source order first (in source order),
  // then any not in source (newly added) in their original order.
  const orderIndex = new Map(sourceOrder.map((name, i) => [name, i] as const));

  const sortByOrder = <T extends { shortName: string }>(items: readonly T[]): T[] => {
    return [...items].sort((a, b) => {
      const ai = orderIndex.get(a.shortName);
      const bi = orderIndex.get(b.shortName);
      if (ai !== undefined && bi !== undefined) return ai - bi;
      if (ai !== undefined) return -1;
      if (bi !== undefined) return 1;
      return 0;
    });
  };

  return {
    ...doc,
    packages: doc.packages.map((p) => ({
      ...p,
      elements: sortByOrder(p.elements),
    })),
  };
}
```

```ts
import { findByPath, type ElementPath } from './path.js';

/**
 * Remove the container at `path` AND any inbound references that target
 * it. The scan is iterative (BFS from the target) — not recursive — to
 * defend against reference cycles. Returns:
 *   - ok:    new doc with target + inbound refs removed
 *   - err:   path-not-found
 *
 * The result doc is the same reference if no changes occurred
 * (no-op convention; see mutation.ts header).
 */
export function removeWithCascade(
  doc: ArxmlDocument,
  path: ElementPath,
): Result<ArxmlDocument, MutationError> {
  // Step 1: confirm target exists
  const target = findByPath(doc, path);
  if (target === null) {
    return err({ kind: 'path-not-found', path });
  }

  // Step 2: find all paths that have a <REFERENCE> targeting `path`
  const inboundRefs = findInboundReferences(doc, path);

  // Step 3: remove target + each inbound ref
  let next = doc;
  next = removeAtPathInternal(next, path);
  for (const refPath of inboundRefs) {
    const removed = removeAtPathInternal(next, refPath);
    if (removed !== null) next = removed;
  }

  return ok(next);
}

function findInboundReferences(doc: ArxmlDocument, target: string): readonly string[] {
  const result: string[] = [];
  walkForReferences(doc, target, result, new Set());
  return result;
}

function walkForReferences(
  doc: ArxmlDocument,
  target: string,
  out: string[],
  visited: Set<string>,
): void {
  // Iterative walk to avoid recursion blow-up on cyclic graphs
  const stack: ArxmlElement[] = doc.packages.flatMap((p) => p.elements);
  while (stack.length > 0) {
    const el = stack.pop()!;
    if (visited.has(el.path)) continue;
    visited.add(el.path);

    if (el.kind === 'reference' && el.target === target) {
      out.push(el.path);
      // Don't recurse into the reference's children (it has none)
      continue;
    }

    if (el.kind === 'module' || el.kind === 'container') {
      stack.push(...el.children);
    }
  }
}

function removeAtPathInternal(doc: ArxmlDocument, path: ElementPath): ArxmlDocument | null {
  // Reuse existing removeAtPath, but return null on not-found instead of Result
  const r = removeAtPath(doc, path);
  return r.kind === 'ok' ? r.value : null;
}
```

```ts
// shared/normalized-document.ts
// v1.5.1 → v1.6.0 interface contract.
//
// NormalizedDocument is the abstraction that unifies DOM-parser output
// and the new streaming reader (PR(6)). All store actions operate on
// NormalizedDocument, not on raw XML strings. The arxml-stream router
// (Task 6) routes to DOM or streaming path based on file size +
// feature flag, but the output type is always this interface.

import type { ArxmlDocument } from '../core/arxml/types.js';

export interface NormalizedDocument {
  /** ARXML schema version extracted from <AR-PACKAGES> root attribute */
  readonly version: '4.x' | '5.x' | '6.x';
  /** Top-level <AR-PACKAGE> elements, in source order (PR(2) preserveSourceOrder) */
  readonly packages: ReadonlyArray<NormalizedPackage>;
  /** Flattened <ECUC-MODULE-DEF-VALUES> for quick iteration */
  readonly modules: ReadonlyArray<NormalizedModule>;
  /** Flattened <REFERENCE> table (target → source) for O(1) lookup */
  readonly references: ReadonlyArray<NormalizedReference>;
  /** Ordered list of all container paths in source order (preserveSourceOrder input) */
  readonly sourceOrder: ReadonlyArray<string>;
  /** Which reader produced this doc (diagnostics) */
  readonly origin: 'dom' | 'stream';
}

export interface NormalizedPackage {
  readonly shortName: string;
  readonly path: string;
  readonly elements: ReadonlyArray<NormalizedElement>;
}

export type NormalizedElement =
  | {
      readonly kind: 'module';
      readonly shortName: string;
      readonly path: string;
      readonly children: ReadonlyArray<NormalizedElement>;
    }
  | {
      readonly kind: 'container';
      readonly shortName: string;
      readonly path: string;
      readonly children: ReadonlyArray<NormalizedElement>;
    }
  | {
      readonly kind: 'reference';
      readonly shortName: string;
      readonly path: string;
      readonly target: string;
    }
  | { readonly kind: 'unknown'; readonly shortName: string; readonly path: string };

export interface NormalizedModule {
  readonly shortName: string;
  readonly path: string;
  readonly definitionRef: string;
}

export interface NormalizedReference {
  readonly source: string; // path of the <REFERENCE> element
  readonly target: string; // DEST attribute value
}

/** Convert an ArxmlDocument to a NormalizedDocument. Used by DOM path. */
export function fromArxmlDocument(
  doc: ArxmlDocument,
  origin: 'dom' | 'stream' = 'dom',
): NormalizedDocument {
  const packages: NormalizedPackage[] = [];
  const modules: NormalizedModule[] = [];
  const references: NormalizedReference[] = [];
  const sourceOrder: string[] = [];

  function walkElement(el: ArxmlElement): NormalizedElement {
    sourceOrder.push(el.path);
    if (el.kind === 'reference') {
      references.push({ source: el.path, target: el.target });
      return { kind: 'reference', shortName: el.shortName, path: el.path, target: el.target };
    }
    if (el.kind === 'unknown') {
      return { kind: 'unknown', shortName: el.shortName, path: el.path };
    }
    // module or container
    if (el.kind === 'module') {
      modules.push({ shortName: el.shortName, path: el.path, definitionRef: el.definitionRef });
    }
    return {
      kind: el.kind,
      shortName: el.shortName,
      path: el.path,
      children: el.children.map(walkElement),
    } as NormalizedElement;
  }

  for (const pkg of doc.packages) {
    sourceOrder.push(pkg.path);
    packages.push({
      shortName: pkg.shortName,
      path: pkg.path,
      elements: pkg.elements.map(walkElement),
    });
  }

  return {
    version: '6.x', // TODO: extract from AR-PACKAGES root attribute
    packages,
    modules,
    references,
    sourceOrder,
    origin,
  };
}
```

```ts
import { promises as fs } from 'node:fs';
import { dirname, join } from 'node:path';

/**
 * Atomic file write. Writes content to `<file>.tmp`, fsyncs, then renames
 * to the final path. On any failure, the temp file is cleaned up and the
 * original file is left untouched. Windows uses `MoveFileEx` (via
 * `fs.rename` which delegates) with `MOVEFILE_REPLACE_EXISTING` (the
 * default behavior of fs.rename on modern Node).
 *
 * Trust-sprint invariant (v1.4.0): never partial-write a user's project
 * file. This is the single point of truth for project file persistence.
 */
export async function writeAtomic(file: string, content: string): Promise<void> {
  const tmp = `${file}.tmp-${process.pid}-${Date.now()}`;
  try {
    await fs.mkdir(dirname(file), { recursive: true });
    await fs.writeFile(tmp, content, 'utf-8');
    const fh = await fs.open(tmp, 'r+');
    try {
      await fh.sync();
    } finally {
      await fh.close();
    }
    await fs.rename(tmp, file);
  } catch (err) {
    // Clean up temp on any failure
    try {
      await fs.unlink(tmp);
    } catch {
      // Ignore: temp may not exist if writeFile failed early
    }
    throw err;
  }
}
```

```ts
// arxml-stream/feature-flag.ts
// Feature flags for the new ARXML streaming + IndexedDB cache layer.
// Both default OFF. To opt-in, set in app config (e.g. via env or
// settings.json). When OFF, the router falls back to the existing
// DOMParser path — behavior is identical to v1.5.0.

import { readFileSync, existsSync } from 'node:fs';
import { join } from 'node:path';

interface FlagsConfig {
  experimental: {
    streaming: boolean;
    indexedDb: boolean;
  };
}

let cached: FlagsConfig | null = null;

function loadFlags(): FlagsConfig {
  if (cached !== null) return cached;
  // Path to settings file: app.getPath('userData')/settings.json
  // We resolve at runtime to avoid hardcoding the userData path
  const candidates = [
    join(process.cwd(), 'settings.json'),
    join(process.env.APPDATA ?? '', 'claude-autosarcfg', 'settings.json'),
  ];
  for (const candidate of candidates) {
    if (existsSync(candidate)) {
      try {
        const raw = readFileSync(candidate, 'utf-8');
        const parsed = JSON.parse(raw) as Partial<FlagsConfig>;
        cached = {
          experimental: {
            streaming: parsed.experimental?.streaming ?? false,
            indexedDb: parsed.experimental?.indexedDb ?? false,
          },
        };
        return cached;
      } catch {
        // Malformed settings; fall through to defaults
      }
    }
  }
  cached = { experimental: { streaming: false, indexedDb: false } };
  return cached;
}

export function isStreamingEnabled(): boolean {
  return loadFlags().experimental.streaming;
}

export function isIndexedDbEnabled(): boolean {
  return loadFlags().experimental.indexedDb;
}

/** For tests: reset the cache. */
export function _resetFlagCache(): void {
  cached = null;
}
```

```ts
// arxml-stream/normalize/output.ts
// Adapter: SAX event stream → NormalizedDocument.
// Used by the streaming reader (Task 7).

import type {
  NormalizedDocument,
  NormalizedElement,
  NormalizedPackage,
  NormalizedModule,
  NormalizedReference,
} from '../../../shared/normalized-document.js';

interface SaxEvent {
  type: 'open' | 'text' | 'close' | 'cdata';
  tagName?: string;
  value?: string;
  attributes?: Record<string, string>;
}

export function buildFromSaxEvents(events: readonly SaxEvent[]): NormalizedDocument {
  // Stack-based parser
  const packages: NormalizedPackage[] = [];
  const modules: NormalizedModule[] = [];
  const references: NormalizedReference[] = [];
  const sourceOrder: string[] = [];

  type StackEntry = {
    kind: 'package' | NormalizedElement;
    path: string;
    children: NormalizedElement[];
  };
  const stack: StackEntry[] = [];

  for (const event of events) {
    if (event.type === 'open' && event.tagName !== undefined) {
      const shortName = event.attributes?.['SHORT-NAME'] ?? '';
      const path = computePath(stack, shortName);

      if (event.tagName === 'AR-PACKAGE') {
        stack.push({ kind: 'package', path, children: [] });
        sourceOrder.push(path);
      } else if (
        event.tagName === 'ECUC-MODULE-DEF-VALUES' ||
        event.tagName === 'ECUC-MODULE-CONFIGURATION-VALUES'
      ) {
        const defRef = event.attributes?.['DEFINITION-REF'] ?? '';
        const el: NormalizedElement = {
          kind: 'module',
          shortName,
          path,
          children: [],
        };
        pushChild(stack, el);
        modules.push({ shortName, path, definitionRef: defRef });
        sourceOrder.push(path);
        stack.push({ kind: el, path, children: el.children });
      } else if (event.tagName === 'REFERENCE') {
        const target = event.attributes?.['DEST'] ?? '';
        const el: NormalizedElement = {
          kind: 'reference',
          shortName,
          path,
          target,
        };
        pushChild(stack, el);
        references.push({ source: path, target });
        sourceOrder.push(path);
      } else {
        // Generic container or unknown
        const el: NormalizedElement = {
          kind: 'container',
          shortName,
          path,
          children: [],
        };
        pushChild(stack, el);
        sourceOrder.push(path);
        stack.push({ kind: el, path, children: el.children });
      }
    } else if (event.type === 'close') {
      const popped = stack.pop();
      if (popped && popped.kind !== 'package') {
        // Replace the placeholder children with the populated array
        if (popped.kind.kind === 'module' || popped.kind.kind === 'container') {
          // Update the parent with the new children array
          updateChildInPlace(stack, popped.path, popped.children as NormalizedElement[]);
        }
      }
    }
  }

  return {
    version: '6.x',
    packages,
    modules,
    references,
    sourceOrder,
    origin: 'stream',
  };
}

function computePath(stack: ReadonlyArray<{ path: string }>, shortName: string): string {
  if (stack.length === 0) return `/${shortName}`;
  return `${stack[stack.length - 1]!.path}/${shortName}`;
}

function pushChild(
  stack: ReadonlyArray<{ kind: 'package' | NormalizedElement; children: NormalizedElement[] }>,
  child: NormalizedElement,
): void {
  if (stack.length === 0) return;
  stack[stack.length - 1]!.children.push(child);
}

function updateChildInPlace(
  stack: ReadonlyArray<{ kind: 'package' | NormalizedElement; children: NormalizedElement[] }>,
  path: string,
  newChildren: NormalizedElement[],
): void {
  // Walk the stack to find the placeholder and replace
  for (let i = stack.length - 1; i >= 0; i--) {
    const entry = stack[i]!;
    if (entry.kind === 'package') continue;
    const idx = entry.children.findIndex((c) => c.path === path);
    if (idx >= 0) {
      const updated = { ...entry.children[idx]!, children: newChildren } as NormalizedElement;
      entry.children[idx] = updated;
      return;
    }
  }
}
```

```ts
// arxml-stream/router.ts
// Path router: dispatches to DOM or streaming parser based on file size
// + feature flag. Both paths produce NormalizedDocument.

import { promises as fs } from 'node:fs';

import { parseArxml } from '../../core/arxml/parser.js';
import { fromArxmlDocument } from '../../shared/normalized-document.js';
import { isStreamingEnabled, isIndexedDbEnabled } from './feature-flag.js';
import { streamParse } from './streaming/index.js';
import { cacheGet } from './cache/indexeddb-store.js';

const STREAMING_THRESHOLD_BYTES = 2 * 1024 * 1024; // 2 MiB

export async function routeArxmlReader(
  file: string,
  content?: string,
): Promise<NormalizedDocument> {
  const data = content ?? (await fs.readFile(file, 'utf-8'));
  const size = Buffer.byteLength(data, 'utf-8');

  // Cache hit (if enabled)
  if (isIndexedDbEnabled()) {
    const cached = await cacheGet({
      filePath: file,
      mtime: 0, // TODO: get from fs.stat
      contentHash: simpleHash(data),
    });
    if (cached !== null) return cached;
  }

  // Choose path
  let doc: NormalizedDocument;
  if (size < STREAMING_THRESHOLD_BYTES) {
    // Small file: always use DOM (fast, no streaming overhead)
    const parsed = parseArxml(data);
    if (parsed.kind !== 'ok') {
      throw new Error(`parse failed: ${parsed.error.message}`);
    }
    doc = fromArxmlDocument(parsed.value, 'dom');
  } else if (isStreamingEnabled()) {
    // Large file + flag on: streaming
    try {
      doc = await streamParse(data);
    } catch (err) {
      // Fallback to DOM on streaming failure (Q6 A)
      console.warn('[arxml-stream] streaming failed, falling back to DOM:', err);
      const parsed = parseArxml(data);
      if (parsed.kind !== 'ok') {
        throw new Error(`parse failed: ${parsed.error.message}`);
      }
      doc = fromArxmlDocument(parsed.value, 'dom');
    }
  } else {
    // Large file + flag off: DOM (no regression for v1.5.0 behavior)
    const parsed = parseArxml(data);
    if (parsed.kind !== 'ok') {
      throw new Error(`parse failed: ${parsed.error.message}`);
    }
    doc = fromArxmlDocument(parsed.value, 'dom');
  }

  return doc;
}

function simpleHash(data: string): string {
  // Simple non-cryptographic hash for cache key
  // (Real impl in PR(6) Sub-C uses crypto.subtle)
  let h = 0;
  for (let i = 0; i < data.length; i++) {
    h = ((h << 5) - h + data.charCodeAt(i)) | 0;
  }
  return h.toString(16);
}
```

```ts
// arxml-stream/streaming/sax-reader.ts
// SAX-style reader using fast-xml-parser. Emits a stream of events
// instead of building a full DOM. This is what makes 10MB+ ARXML
// files parse without blocking the main thread (when used with
// chunk-builder).

import { XMLParser } from 'fast-xml-parser';

export type SaxEvent =
  | { type: 'open'; tagName: string; attributes: Record<string, string> }
  | { type: 'text'; value: string }
  | { type: 'close'; tagName: string };

export function createSaxReader(content: string): AsyncIterable<SaxEvent> {
  const parser = new XMLParser({
    ignoreAttributes: false,
    attributeNamePrefix: '',
    parseAttributeValue: false,
    parseTagValue: false,
    trimValues: true,
  });

  return {
    async *[Symbol.asyncIterator]() {
      // fast-xml-parser doesn't have a true streaming mode; we use
      // the standard parseAsync (if available) or chunk the input.
      // For v1.5.1, we use the synchronous parser but yield events
      // so consumers can interleave. v1.7.0 will swap in a true SAX
      // parser (e.g. sax-js) if perf becomes critical.
      const tree = parser.parse(content);

      yield* walkTree(tree);
    },
  };
}

function* walkTree(node: unknown): Generator<SaxEvent> {
  if (node === null || node === undefined) return;
  if (typeof node !== 'object') {
    yield { type: 'text', value: String(node) };
    return;
  }
  if (Array.isArray(node)) {
    for (const child of node) yield* walkTree(child);
    return;
  }
  for (const [tagName, value] of Object.entries(node as Record<string, unknown>)) {
    if (tagName.startsWith('@_')) continue; // attribute (handled differently)
    if (tagName === '#text') {
      if (typeof value === 'string') yield { type: 'text', value };
      continue;
    }
    if (typeof value === 'object' && value !== null) {
      const attrs: Record<string, string> = {};
      for (const [k, v] of Object.entries(value as Record<string, unknown>)) {
        if (k.startsWith('@_')) attrs[k.slice(2)] = String(v);
      }
      yield { type: 'open', tagName, attributes: attrs };
      yield* walkTree((value as Record<string, unknown>)['#text'] ?? value);
      yield { type: 'close', tagName };
    } else {
      yield { type: 'text', value: String(value) };
    }
  }
}
```

```ts
export async function* chunkByLines(content: string, linesPerChunk = 500): AsyncIterable<string> {
  const lines = content.split('\n');
  for (let i = 0; i < lines.length; i += linesPerChunk) {
    yield lines.slice(i, i + linesPerChunk).join('\n');
```

```ts
// arxml-stream/cache/indexeddb-store.ts
// IndexedDB-backed cache for NormalizedDocument.
//
// Note: This runs in the main process (Electron), which doesn't have
// IndexedDB natively. We use `fake-indexeddb` (peer dep) to provide
// the API. In a renderer-context alternative, the same code works
// with the browser's native IndexedDB.

import { IDBPDatabase, openDB } from 'idb';

import type { NormalizedDocument } from '../../../shared/normalized-document.js';
import { CACHE_DB_NAME, CACHE_STORE_NAME, CACHE_VERSION } from './schema-version.js';
import type { CacheKey } from './invalidation.js';

let dbPromise: Promise<IDBPDatabase> | null = null;

async function getDb(): Promise<IDBPDatabase> {
  if (dbPromise === null) {
    dbPromise = openDB(CACHE_DB_NAME, CACHE_VERSION, {
      upgrade(db) {
        if (!db.objectStoreNames.contains(CACHE_STORE_NAME)) {
          const store = db.createObjectStore(CACHE_STORE_NAME, { keyPath: 'key' });
          store.createIndex('filePath', 'key.filePath', { unique: false });
        }
      },
    });
  }
  return dbPromise;
}

function keyToString(key: CacheKey): string {
  return `${key.filePath}::${key.mtime}::${key.contentHash}`;
}

export async function cacheGet(key: CacheKey): Promise<NormalizedDocument | null> {
  const db = await getDb();
  const result = await db.get(CACHE_STORE_NAME, keyToString(key));
  return (result as { doc: NormalizedDocument } | undefined)?.doc ?? null;
}

export async function cacheSet(key: CacheKey, doc: NormalizedDocument): Promise<void> {
  const db = await getDb();
  await db.put(CACHE_STORE_NAME, { key: keyToString(key), doc });
}

/** For tests only. */
export async function _clearForTest(): Promise<void> {
  const db = await getDb();
  await db.clear(CACHE_STORE_NAME);
}

/** For tests only. */
export function _resetForTest(): void {
  dbPromise = null;
}
```

```ts
type Mode = 'dom' | 'dom-preserve-order' | 'stream';

async function parseInMode(content: string, mode: Mode): Promise<unknown> {
  if (mode === 'stream') {
    return streamParse(content);
  }
  const parsed = parseArxml(content);
  if (parsed.kind !== 'ok') throw new Error(`parse failed: ${parsed.error.message}`);
  if (mode === 'dom-preserve-order') {
    return fromArxmlDocument(parsed.value, 'dom');
  }
  return parsed.value;

function serializeInMode(doc: unknown, mode: Mode, source?: string): string {
  if (mode === 'stream') {
    // For stream mode, re-parse via DOM for serialization
    const reparsed = parseArxml(JSON.stringify(doc)); // hack
    throw new Error('stream mode serialize not implemented; use dom-preserve-order');

    // Convert NormalizedDocument back to ArxmlDocument
    // (v1.6.0 will have a proper converter; for now we just use dom mode)
    return serializeArxml(doc as any, { sourceArxml: source }).value!;
  }
  return serializeArxml(doc as any, { sourceArxml: source }).value!;
}

function diffUnderTolerance(a: string, b: string): string | null {
  // Strip namespace prefix order, whitespace, comments, attribute order
  let aNorm = a;
  let bNorm = b;
  if (TOLERANCE_RULES.namespacePrefixOrder) {
    // Sort namespace declarations alphabetically
    aNorm = aNorm.replace(/xmlns(:\w+)?="[^"]*"/g, '');
    bNorm = bNorm.replace(/xmlns(:\w+)?="[^"]*"/g, '');
  }
  if (TOLERANCE_RULES.whitespace) {
    aNorm = aNorm.replace(/\s+/g, ' ').trim();
    bNorm = bNorm.replace(/\s+/g, ' ').trim();
  }
  if (TOLERANCE_RULES.comments) {
    aNorm = aNorm.replace(/<!--[\s\S]*?-->/g, '');
    bNorm = bNorm.replace(/<!--[\s\S]*?-->/g, '');
  }
  if (TOLERANCE_RULES.attributeOrder) {
    // Sort attributes within each element
    aNorm = aNorm.replace(/<(\w+)\s+([^>]+?)\s*\/?>/g, (_, tag, attrs) => {
      const sorted = attrs.split(/\s+/).filter(Boolean).sort().join(' ');
      return `<${tag} ${sorted}>`;
    });
    bNorm = bNorm.replace(/<(\w+)\s+([^>]+?)\s*\/?>/g, (_, tag, attrs) => {
      const sorted = attrs.split(/\s+/).filter(Boolean).sort().join(' ');
      return `<${tag} ${sorted}>`;
    });
  }
  return aNorm === bNorm
    ? null
    : `Diff after tolerance:\nA: ${aNorm.slice(0, 200)}\nB: ${bNorm.slice(0, 200)}`;
}
```
