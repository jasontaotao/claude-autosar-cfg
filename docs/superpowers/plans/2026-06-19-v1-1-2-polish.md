In `src/shared/path.ts`, replace the already-relative branch (lines 80-88) with:

```ts
  // Already-relative input (no leading '/' and no drive letter): pass
  // through unchanged UNLESS the input contains a parent-traversal
  // segment. We deliberately do NOT call `path.posix.normalize` here
  // (renderer-safe, no node:path); instead we reject any path whose
  // segments contain `..` after splitting on `/` and stripping empty
  // segments. The caller can then treat `null` identically to
  // "outside manifestDir" (typically: keep absolute path, surface
  // error on next save round-trip).
  if (dirDrive === undefined && fileDrive === undefined && !filePath.startsWith('/')) {
    const segments = filePath.replace(/\\/g, '/').split('/').filter((s) => s !== '');
    if (segments.some((s) => s === '..')) return null;
    return filePath;
  }
  return null;
```

Also update the JSDoc comment block at lines 47-51 to add the rejection case:

```ts
 * Rejected inputs (return `null`):
 *   - Cross-drive Windows: `D:\proj` + `E:\proj\X.arxml`
 *   - Sibling POSIX:       `/a/b` + `/a/c/X.arxml` (different parent)
 *   - Asymmetric drive prefix (one has, one doesn't)
 *   - Empty `filePath`
 *   - Already-relative input containing `..` segment (parent traversal)
```

```ts
  readonly message: string;

  readonly autoDismissMs?: number;
}

interface ArxmlStoreState {
  // ... existing fields ...
  toast: ToastState | null;  // new field; replaces single `error: string | null`
  setInfo: (message: string, autoDismissMs?: number) => void;
  setSuccess: (message: string, autoDismissMs?: number) => void;
  setWarning: (message: string, autoDismissMs?: number) => void;
  dismissToast: () => void;  // replaces setError(null)

// Migration: keep `error: string | null` as a getter that returns
// `toast?.message ?? null` for backward compat with existing callers.
// setError remains, but routes to toast with kind='error'.

setError: (msg: string | null) => set({ toast: msg === null ? null : { kind: 'error', message: msg } }),
setInfo: (msg, autoDismissMs = 3000) => set({ toast: { kind: 'info', message: msg, autoDismissMs } }),
setSuccess: (msg, autoDismissMs = 3000) => set({ toast: { kind: 'success', message: msg, autoDismissMs } }),
setWarning: (msg, autoDismissMs = 5000) => set({ toast: { kind: 'warning', message: msg, autoDismissMs } }),
dismissToast: () => set({ toast: null }),
```

```ts
import { findByPathMultiDoc } from '@core/arxml/path.js';

export interface ResolvedContainerTarget {
  readonly doc: ArxmlDocument;
  readonly filePath: string;
  readonly innerPath: string;
}

/**
 * Resolve a containerPath to its source document in either view mode.
 * - 'combined' mode: routes via findByPathMultiDoc (path may include
 *   the source file's basename as a prefix).
 * - 'single' mode: returns the active doc; innerPath == containerPath.
 * - Returns null when no source can be resolved (caller treats as
 *   'path-not-found' error).
 */
export function resolveContainerTarget(
  state: ArxmlStoreState,
  containerPath: string,
): ResolvedContainerTarget | null {
  if (state.viewMode === 'combined') {
    const hit = findByPathMultiDoc(state.documents, state.documentPaths, containerPath);
    if (hit === null) return null;
    return { doc: hit.doc, filePath: hit.filePath, innerPath: containerPath };
  }
  if (state.doc === null) return null;
  return { doc: state.doc, filePath: state.docPath ?? '', innerPath: containerPath };
}
```
