```ts
// src/core/validation/validate.ts (add new exported function)
import type { BswModuleDef } from '../project/bswmd.js';

export interface VariantCoverageWarning {
  readonly kind: 'BSW-SEC-005';
  readonly severity: 'error';
  readonly message: string;
  readonly path: string;
}

/**
 * BSW-SEC-005 (v1.17.0) — POST-BUILD parameter without variant coverage.
 *
 * Walks every container's parameters, collects those whose
 * multiplicityConfigClasses include `POST-BUILD`, and checks that at
 * least one runtime value carries a variantRef covering the path.
 * Recurses into subContainers + choices (same traversal pattern as
 * `findContainerInTree` in bswmd.ts).
 */
export function validateVariantCoverage(
  values: ReadonlyArray<{ readonly paramPath: string; readonly variantRef?: string }>,
  bswmd: BswModuleDef,
): ReadonlyArray<VariantCoverageWarning> {
  const warnings: VariantCoverageWarning[] = [];
  const visit = (c: {
    readonly path: string;
    readonly parameters: readonly { readonly path: string }[];
    readonly multiplicityConfigClasses?: readonly { readonly configClass: string }[];
    readonly subContainers: readonly unknown[];
    readonly choices: readonly unknown[];
  }): void => {
    const isPostBuild =
      c.multiplicityConfigClasses?.some((m) => m.configClass === 'POST-BUILD') ?? false;
    if (isPostBuild) {
      for (const p of c.parameters) {
        const covered = values.some((v) => v.paramPath === p.path && v.variantRef !== undefined);
        if (!covered) {
          warnings.push({
            kind: 'BSW-SEC-005',
            severity: 'error',
            message: `POST-BUILD parameter ${p.path} requires variant coverage`,
            path: p.path,
          });
        }
      }
    }
    for (const sub of c.subContainers) visit(sub as Parameters<typeof visit>[0]);
    for (const ch of c.choices) visit(ch as Parameters<typeof visit>[0]);
  };
  for (const c of bswmd.containers) visit(c);
  return warnings;
}
```

// src/core/project/bswmd.ts (modify parseBswmd / walkPackagesForModules)

// Locate the existing walkPackagesForModules function and add a parallel
// walk that collects <MODULE-REF> children. Strategy:
//   1. Define walkPackagesForModuleRefs(node, parentPath, out) that
//      walks AR-PACKAGES recursively (same as walkPackagesForModules)
//      and inside each AR-PACKAGE/ELEMENTS, looks for <MODULE-REF>
//      children (e.g. under <ECUC-DEFINITION-COLLECTION>).
//   2. Each <MODULE-REF> has a DEST attribute and a text body
//      (target path). Surface as ModuleRefEntry.
//   3. After walkPackagesForModules returns, call walkPackagesForModuleRefs
//      and attach the result to BswmdDocument.moduleRefs.

const moduleRefs: ModuleRefEntry[] = [];
walkPackagesForModuleRefs(arPackages as Record<string, unknown>, '', moduleRefs);

return { ok: true, value: { version, modules, warnings, moduleRefs } };

// Where walkPackagesForModuleRefs is a new helper that mirrors
// walkPackagesForModules but extracts <MODULE-REF> elements:

function walkPackagesForModuleRefs(
  node: Record<string, unknown>,
  parentPath: string,
  out: ModuleRefEntry[],
): void {
  for (const pkg of asArray<Record<string, unknown>>(node['AR-PACKAGE'])) {
    const shortName = readShortName(pkg);
    if (shortName === undefined) continue;
    const path = `${parentPath}/${shortName}`;
    const elementsRaw = pkg['ELEMENTS'];
    if (typeof elementsRaw === 'object' && elementsRaw !== null) {
      for (const item of asArray<Record<string, unknown>>(
        (elementsRaw as Record<string, unknown>)['MODULE-REF'] ?? [],
      )) {
        const targetPath = readElementText(item);
        if (targetPath !== '') {
          out.push({ target: targetPath, source: path });
        }
      }
    }
    const nestedRaw = pkg['AR-PACKAGES'];
    if (typeof nestedRaw === 'object' && nestedRaw !== null) {
      walkPackagesForModuleRefs(nestedRaw as Record<string, unknown>, path, out);
    }
  }
}
