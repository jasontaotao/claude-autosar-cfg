```ts
// src/core/validation/validate.ts (add new exported function)
import type { BswmdModuleDef, EcucValue } from '../project/bswmd.js';

export interface VariantCoverageWarning {
  readonly kind: 'BSW-SEC-005';
  readonly severity: 'error';
  readonly message: string;
  readonly path: string;
}

export function validateVariantCoverage(
  values: ReadonlyArray<EcucValue>,
  bswmd: BswmdModuleDef,
): ReadonlyArray<VariantCoverageWarning> {
  const warnings: VariantCoverageWarning[] = [];
  for (const param of bswmd.parameters) {
    if (param.multiplicity?.configClass !== 'POST-BUILD') continue;
    const covered = values.some(
      (v) => v.paramPath === `${bswmd.name}/${param.name}` && v.variantRef !== undefined,
    );
    if (!covered) {
      warnings.push({
        kind: 'BSW-SEC-005',
        severity: 'error',
        message: `POST-BUILD parameter ${bswmd.name}/${param.name} requires variant coverage`,
        path: `${bswmd.name}/${param.name}`,
      });
    }
  }
  return warnings;
}
```
