## 2. Architecture

### 2.1 T1: Real headless dispatcher

**Edit `src/main/ipc/headless-stubs.ts`** — replace `headlessRunCommandStub` with real dispatcher:

```ts
// v1.19.0 MINOR — real headless dispatcher.
// Replaces headlessRunCommandStub. Delegates to src/cli/command-dispatcher.ts.
// Emits MutateAppliedEvent / ValidateResultEvent to the renderer after
// mutate / validate commands complete (so renderer consumers get
// notified regardless of whether command was triggered via CLI or GUI).

import type { IpcMainInvokeEvent } from 'electron';

import { dispatchCommand } from '../../../cli/command-dispatcher.js';
import type {
  HeadlessRunCommandRequest,
  HeadlessRunCommandResult,
} from '../../../shared/headless/ipc-contract.js';

import { emitMutateApplied, emitValidateResult } from '../headless/push-emitters.js';

export async function headlessRunCommandHandler(
  _evt: IpcMainInvokeEvent,
  req: HeadlessRunCommandRequest,
): Promise<HeadlessRunCommandResult> {
  const result = await dispatchCommand(req.parsedArgs);

  // Emit push events for mutate / validate commands so the renderer
  // consumer (T3) gets notified.
  if (result.kind === 'mutate') {
    emitMutateApplied({
      patchId: req.patchId,
      applied: result.applied,
      warnings: result.warnings,
    });
  } else if (result.kind === 'validate') {
    emitValidateResult(result.validateEnvelope);
  }

  return { kind: 'ok', result };
}
```

const closeProject = useCallback(async (): Promise<ProjectActionResult> => {
  // v1.19.0 MINOR — wire v1.18.2 PROJECT_CLOSE IPC. Symmetric counterpart
  // to openProjectFromDialog. No dirty guard (the closeProject user
  // gesture is "I want to start over" — unsaved-changes flow handled

    // preserves documents[] and dirtyPaths so the user keeps editing
    // in loose mode.
    useArxmlStore.getState().closeProject();
    return { kind: 'ok' };
  }
  return { kind: 'ok' }; // idempotent
}, []);

- **`useArxmlStore.closeProject` action**: per CHANGELOG line 1543-1544 (v1.6.0-era), the store already has a `closeProject` action that preserves `documents[]` and `dirtyPaths`. Verify by reading the store.
- **SCRIPT_PROGRESS IPC channel preload**: needs to verify the preload bridge exposes the channel. If missing, add it.
