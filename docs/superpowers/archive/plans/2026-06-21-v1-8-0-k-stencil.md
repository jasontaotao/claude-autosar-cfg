# v1.8.0 — K Stencil Wizard Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a GUI wizard that generates minimal valid ECUC module skeletons (Com / ComM / PduR / EcuC) without requiring BSWMD, optionally gated by the v1.6.0 Cluster G SWS Validator. Output is a single `.arxml` file the user can save anywhere and reopen as a template.

**Architecture:** Renderer modal triggers new IPC channel `stencil:generate:v1`. Main process loads family-specific schema, optionally invokes G's `sws-validator:run:v1` (v1.6.0), serializes result back as `.arxml` string. With-BSWMD mode merges via existing `applyPatchSteps` (v1.5.1 A+C). Feature flag `experimental.stencilWizard` default OFF. CLI parity deferred to v1.9.0+.

**Tech Stack:** Electron 30 + TypeScript 5 strict + React 18 + Zustand 4 + Vitest 1 + Playwright 1.45. Reuses `sws-validator:run:v1` (v1.6.0 G) and `applyPatchSteps` (v1.5.1 A+C) verbatim.

import { readFileSync, existsSync } from 'node:fs';
import { join } from 'node:path';

let cached: boolean | null = null;
export function isStencilWizardEnabled(): boolean {
  if (cached !== null) return cached;
  try {
    const candidates = [
      join(process.cwd(), 'settings.json'),
      join(process.env.APPDATA ?? '', 'claude-autosarcfg', 'settings.json'),
    ];
    for (const path of candidates) {
      if (!existsSync(path)) continue;
      const raw = readFileSync(path, 'utf-8');
      const parsed = JSON.parse(raw) as { experimental?: { stencilWizard?: boolean } };
      cached = parsed.experimental?.stencilWizard ?? false;
      return cached;
    }
  } catch { /* fall through */ }
  cached = false;
  return false;
}
export function _resetStencilFlagCache(): void { cached = null; }

// src/main/stencil/schemas/com.ts
// AUTOSAR SWS_Com reference — minimal valid Com module skeleton
// Container hierarchy derived from SWS_Com §8 Container Definitions
//
// CORRECTED after Task 2 review: ArxmlDocument shape uses
// packages[].elements[] (kind: 'module'), not rootPackages[].containers[].
// Container nodes must carry tagName: 'ECUC-CONTAINER-VALUE' (Sprint 14
// bug-2a fix in src/core/arxml/skeleton.ts:127).
import type { ArxmlDocument, ArxmlPackage, ArxmlElement, ArxmlContainer, ArxmlParam } from '../../../core/arxml/types.js';

function param(shortName: string, value: ArxmlParam['value']): ArxmlParam {
  return { shortName, value } as ArxmlParam;
}

export function buildComModule(): ArxmlDocument {
  const comConfig: ArxmlContainer = {
    kind: 'container',
    tagName: 'ECUC-CONTAINER-VALUE',
    shortName: 'ComConfig',
    containers: [],
    params: [param('ComConfigurationClass', { type: 'enum', value: 'PRE_COMPILE' })],
  };
  comConfig.containers.push({
    kind: 'container',
    tagName: 'ECUC-CONTAINER-VALUE',
    shortName: 'ComIPdu',
    containers: [],
    params: [
      param('ComPduId', { type: 'integer', value: 0 }),
      param('ComIPduDirection', { type: 'enum', value: 'RECEIVE' }),
    ],
  });
  const comModule: ArxmlElement = {
    kind: 'module',
    tagName: 'AR-PACKAGE',
    shortName: 'Com',
    containers: [comConfig],
    params: [],
  };
  const comPackage: ArxmlPackage = {
    shortName: 'Com',
    path: '/Com',
    elements: [comModule],
  };
  return { packages: [comPackage] };
}
```

// src/main/ipc/stencilHandler.ts
import { ipcMain } from 'electron';
import { buildStencil } from '../stencil/builder.js';
import { serializeArxml } from '../arxml/serializer.js';
import type { StencilRequest, StencilResponse, StencilFamily } from '../stencil/types.js';

const FILENAMES: Record<StencilFamily, string> = {
  com: 'Com.arxml', comm: 'ComM.arxml', pdur: 'PduR.arxml', ecuc: 'EcuC.arxml',
};

export async function handleStencilGenerate(req: StencilRequest): Promise<StencilResponse> {
  try {
    const doc = buildStencil(req.family);
    const xml = serializeArxml(doc);
    return { ok: true, xml, suggestedFilename: FILENAMES[req.family] };
  } catch (e) {
    return { ok: false, error: { code: 'BUILD_FAILED', i18nKey: 'stencil.error.buildFailed' } };
  }
}

// Gate integration added in Task 8; CLI parity in v1.9.0+
export function registerStencilHandler(): void {
  ipcMain.handle('stencil:generate:v1', async (_event, req: StencilRequest) => handleStencilGenerate(req));
}
```
