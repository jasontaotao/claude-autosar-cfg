```ts
import type { ArxmlDocument, ArxmlElement } from './types';
import type { BswmdDocument, BswModuleDef } from '../project/bswmd';

/**
 * Sprint 14 — generate a value-side ECUC skeleton from a BSWMD module.
 *
 * The output document is a minimal `<ECUC-MODULE-CONFIGURATION-VALUES>`
 * skeleton: SHORT-NAME, MODULE-REF (pointing back to the BSWMD module path),
 * and a CONTAINERS block with one empty `<ECUC-CONFIGURATION-CONTAINER>`
 * per container in the BSWMD definition. PARAMETERS and REFERENCES blocks
 * are emitted but empty (the user fills them in via ParamEditor after creation).
 *
 * Pure: no I/O, no React, no Zustand. Caller attaches `path` and
 * `sourceBswmdPath` to the returned document.
 *
 * @throws if `moduleShortName` is not found in `doc.modules`.
 */
export function generateEcucSkeleton(doc: BswmdDocument, moduleShortName: string): ArxmlDocument {
  const mod = doc.modules.find((m) => m.shortName === moduleShortName);
  if (mod === undefined) {
    throw new Error(`BSWMD module "${moduleShortName}" not found in document`);
  }
  return {
    path: '',
    root: buildRoot(mod),
  };
}

function buildRoot(mod: BswModuleDef): ArxmlElement {
  return {
    tagName: 'ECUC-MODULE-CONFIGURATION-VALUES',
    attributes: {
      xmlns: 'http://autosar.org/schema/r4.0',
      'xmlns:xsi': 'http://www.w3.org/2001/XMLSchema-instance',
    },
    children: [
      { tagName: 'SHORT-NAME', attributes: {}, children: [], text: mod.shortName },
      {
        tagName: 'MODULE-REF',
        attributes: { DEST: 'ECUC-MODULE-DEF' },
        children: [],
        text: mod.path,
      },
      { tagName: 'CONTAINERS', attributes: {}, children: mod.containers.map(buildContainer) },
      { tagName: 'PARAMETERS', attributes: {}, children: [] },
      { tagName: 'REFERENCES', attributes: {}, children: [] },
    ],
  };
}

function buildContainer(c: BswModuleDef['containers'][number]): ArxmlElement {
  return {
    tagName: 'ECUC-CONFIGURATION-CONTAINER',
    attributes: {},
    children: [
      { tagName: 'SHORT-NAME', attributes: {}, children: [], text: c.shortName },
      { tagName: 'SUB-CONTAINERS', attributes: {}, children: c.subContainers.map(buildContainer) },
      { tagName: 'PARAMETER-VALUES', attributes: {}, children: [] },
      { tagName: 'REFERENCE-VALUES', attributes: {}, children: [] },
    ],
  };
}
```

```ts
export interface PickedModule {
  readonly bswmdPath: string;
  readonly moduleShortName: string;
}

/**
 * Compute final file paths for a batch of PickedModule, applying collision
 * resolution: when N ≥ 2 picks share the same `moduleShortName`, the second
 * (and later) are suffixed with `__<vendorKey>` where vendorKey comes from
 * the BSWMD file basename. If basenames collide too, fall back to numeric
 * suffix (`__can_1`, `__can_2`).
 *
 * Pure. The Map key is `${moduleShortName}/${bswmdPath}` (always unique
 * since bswmdPath is unique per file).
 */
export function resolveCollisionFilename(
  picks: readonly PickedModule[],
  projectDir: string,
): Map<string, string> {
  const out = new Map<string, string>();
  // Group picks by moduleShortName; first pick in iteration order wins
  // the un-suffixed name, others get vendor suffix.
  const groups = new Map<string, PickedModule[]>();
  for (const p of picks) {
    const list = groups.get(p.moduleShortName) ?? [];
    list.push(p);
    groups.set(p.moduleShortName, list);
  }
  for (const [, group] of groups) {
    if (group.length === 1) {
      const p = group[0]!;
      out.set(key(p), `${projectDir}/${p.moduleShortName}_Cfg.arxml`);
      continue;
    }
    // Compute vendor keys per pick; if duplicates, fall back to numeric.
    const vendorKeys = group.map((p) => vendorKeyFromPath(p.bswmdPath));
    const seen = new Map<string, number>();
    group.forEach((p, idx) => {
      const baseKey = vendorKeys[idx]!;
      const seenCount = seen.get(baseKey) ?? 0;
      seen.set(baseKey, seenCount + 1);
      const suffix = seenCount === 0 ? '' : `_${seenCount}`;
      const vendorPart = seenCount === 0 ? baseKey : `${baseKey}${suffix}`;
      const fileName =
        seenCount === 0
          ? `${p.moduleShortName}_Cfg.arxml`
          : `${p.moduleShortName}__${vendorPart}_Cfg.arxml`;
      out.set(key(p), `${projectDir}/${fileName}`);
    });
  }
  return out;
}

function key(p: PickedModule): string {
  return `${p.moduleShortName}/${p.bswmdPath}`;
}

function vendorKeyFromPath(p: string): string {
  // Extract basename without ".arxml", lowercased.
  const basename = p.split(/[\\/]/).pop() ?? p;
  return basename.replace(/\.arxml$/i, '').toLowerCase();
}
```

import { promises as fs } from 'node:fs';
import path from 'node:path';
import type {
  ProjectWriteArxmlBatchRequest,
  ProjectWriteArxmlBatchResult,
} from '../../shared/types';

export async function projectWriteArxmlBatchHandler(
  req: ProjectWriteArxmlBatchRequest,
): Promise<ProjectWriteArxmlBatchResult> {
  const written: string[] = [];
  const failed: { filePath: string; message: string }[] = [];
  for (const f of req.files) {
    try {
      await fs.mkdir(path.dirname(f.filePath), { recursive: true });
      await fs.writeFile(f.filePath, f.content, 'utf-8');
      written.push(f.filePath);
    } catch (err) {
      failed.push({
        filePath: f.filePath,
        message: err instanceof Error ? err.message : String(err),
      });
    }
  }
  if (failed.length === 0) return { kind: 'ok', written };
  if (written.length === 0) return { kind: 'write-failed', message: failed[0]!.message };
  return { kind: 'partial', written, failed };
}
```

import { promises as fs } from 'node:fs';
import type { ProjectDeleteArxmlRequest, ProjectDeleteArxmlResult } from '../../shared/types';

export async function projectDeleteArxmlHandler(
  req: ProjectDeleteArxmlRequest,
): Promise<ProjectDeleteArxmlResult> {
  try {
    await fs.unlink(req.filePath);
    return { kind: 'ok' };
  } catch (err) {
    const code = (err as NodeJS.ErrnoException).code;
    if (code === 'ENOENT') return { kind: 'not-found' };
    return {
      kind: 'write-failed',
      message: err instanceof Error ? err.message : String(err),
    };
  }
}

import { useCallback } from 'react';
import type { PickedModule } from '../../core/arxml/skeleton';
import { generateEcucSkeleton, resolveCollisionFilename } from '../../core/arxml/skeleton';
import { useArxmlStore } from '../store/useArxmlStore';
import type { ProjectWriteArxmlBatchResult } from '../../shared/types';

declare global {
  interface Window {
    autosarApi: {
      writeArxmlBatch: (req: {
        files: { filePath: string; content: string }[];
      }) => Promise<ProjectWriteArxmlBatchResult>;
      deleteArxml: (req: {
        filePath: string;
      }) => Promise<{ kind: 'ok' | 'not-found' | 'write-failed'; message?: string }>;
    };
  }
}

export interface CreateEcucInput {
  readonly picks: readonly PickedModule[];
  readonly projectDir: string;
}

export interface CreateEcucResult {
  readonly kind: 'ok' | 'partial' | 'error';
  readonly written: readonly string[];
  readonly failed: readonly { filePath: string; message: string }[];
  readonly message?: string;
}

export function useCreateEcucFromBswmd(): {
  readonly create: (input: CreateEcucInput) => Promise<CreateEcucResult>;
} {
  const create = useCallback(async (input: CreateEcucInput): Promise<CreateEcucResult> => {
    const state = useArxmlStore.getState();
    const paths = resolveCollisionFilename(input.picks, input.projectDir);
    // Serialize each pick.
    const files: { filePath: string; content: string }[] = [];
    for (const p of input.picks) {
      const schemaIdx = state.bswmdPaths.indexOf(p.bswmdPath);
      const schema = state.bswmdSchemas[schemaIdx];
      if (schema === undefined) continue;
      const skeleton = generateEcucSkeleton(schema, p.moduleShortName);
      // Serialize via existing arxml serializer (kept inline for now — call site can
      // wire to serializer.serializeArxml in a follow-up).
      const content = `<?xml version="1.0" encoding="UTF-8"?>\n${serialize(skeleton.root)}`;
      const filePath = paths.get(`${p.moduleShortName}/${p.bswmdPath}`);
      if (filePath === undefined) continue;
      files.push({ filePath, content });
    }
    const result = await window.autosarApi.writeArxmlBatch({ files });
    if (result.kind === 'write-failed') {
      return { kind: 'error', written: [], failed: [], message: result.message };
    }
    if (result.kind === 'partial') {
      // Roll back partial writes.
      for (const fp of result.written) {
        await window.autosarApi.deleteArxml({ filePath: fp });
      }
      return {
        kind: 'partial',
        written: result.written,
        failed: result.failed,
        message: 'partial batch failure — partial writes rolled back',
      };
    }
    // result.kind === 'ok': add documents + mark dirty.
    for (const p of input.picks) {
      const filePath = paths.get(`${p.moduleShortName}/${p.bswmdPath}`);
      if (filePath === undefined) continue;
      const schemaIdx = state.bswmdPaths.indexOf(p.bswmdPath);
      const schema = state.bswmdSchemas[schemaIdx];
      if (schema === undefined) continue;
      const skeleton = generateEcucSkeleton(schema, p.moduleShortName);
      useArxmlStore.getState().addDocumentWithSource({ ...skeleton, path: filePath }, p.bswmdPath);
      useArxmlStore.getState().markPathDirty(filePath);
    }
    return { kind: 'ok', written: result.written, failed: [] };
  }, []);
  return { create };
}

// Minimal serializer for the skeleton — produces XML string from ArxmlElement tree.
// Calls site can replace with the project's full serializer if it accepts (path, root).
function serialize(
  el: { tagName: string; attributes: Record<string, string>; children: unknown[]; text?: string },
  depth = 0,
): string {
  const indent = '  '.repeat(depth);
  const attrs = Object.entries(el.attributes)
    .map(([k, v]) => ` ${k}="${v}"`)
    .join('');
  if (el.text !== undefined) return `${indent}<${el.tagName}${attrs}>${el.text}</${el.tagName}>`;
  if (el.children.length === 0) return `${indent}<${el.tagName}${attrs}/>`;
  const inner = (
    el.children as {
      tagName: string;
      attributes: Record<string, string>;
      children: unknown[];
      text?: string;
    }[]
  )
    .map((c) => serialize(c, depth + 1))
    .join('\n');
  return `${indent}<${el.tagName}${attrs}>\n${inner}\n${indent}</${el.tagName}>`;
}

export function ModuleFromBswmdPicker({
  open,
  projectDir,
  onConfirm,
  onClose,
  preSelectedBswmdPath,
}: Props): JSX.Element | null {
  const bswmdSchemas = useArxmlStore((s) => s.bswmdSchemas);
  const bswmdPaths = useArxmlStore((s) => s.bswmdPaths);
  const locale = useArxmlStore((s) => s.locale);
  const [selected, setSelected] = useState<Set<string>>(new Set());
  const [filter, setFilter] = useState('');

  const allPicks = useMemo<PickedModule[]>(() => {
    const out: PickedModule[] = [];
    bswmdPaths.forEach((bp, idx) => {
      const schema = bswmdSchemas[idx];
      if (schema === undefined) return;
      schema.modules.forEach((m) => {
        if (filter !== '' && !m.shortName.toLowerCase().includes(filter.toLowerCase())) return;
        out.push({ bswmdPath: bp, moduleShortName: m.shortName });

    () => allPicks.filter((p) => selected.has(`${p.moduleShortName}/${p.bswmdPath}`)),
    [allPicks, selected],

  const collisionMap = useMemo(
    () => resolveCollisionFilename(selectedPicks, projectDir),
    [selectedPicks, projectDir],
  );

      role="dialog"

          </button>

const removeBswmdWithCascade = useCallback(async (path: string): Promise<ProjectActionResult> => {
  const state = useArxmlStore.getState();
  const dependents = state.findDependentsOfBswmd(path);
  if (dependents.length > 0) {
    const answer = await confirm({
      title: '移除 BSWMD',
      message: `BSWMD "${path}" 有 ${dependents.length} 个依赖文件，移除后将失去 schema 校验。`,
      options: [
        { id: 'onlyBswmd', label: '仅移除 BSWMD（保留 ECUC）' },
        { id: 'cascade', label: '移除 + 删除依赖' },
      ],
    });
    if (answer === 'cancel') return { kind: 'ok' };
    if (answer === 'cascade') {
      for (const fp of dependents) {
        await window.autosarApi.deleteArxml({ filePath: fp });
        state.removeDocument(fp);
      }
    }
  }
  state.removeBswmd(path);
  return { kind: 'ok' };
}, []);
```
